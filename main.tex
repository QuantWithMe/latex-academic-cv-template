% Define a custom command for CV section titles.
% The command prints the section title in bold uppercase text,
% adds a small amount of vertical space above it,
% and draws a horizontal rule below it.

\newcommand{\cvsection}[1]{%
    \vspace{5pt}%
    {\large\textbf{\MakeUppercase{#1}}}%
    \vspace{2pt}
    \hrule%
    \vspace{5pt}%
}



% Define a compact CV entry with two aligned rows.
% Arguments:
% #1 = institution or organization
% #2 = location
% #3 = degree, role, or position
% #4 = dates

\newcommand{\cventry}[4]{%
    \vspace{4pt}%
    \noindent
    \begin{tabular*}{\textwidth}{@{}l@{\extracolsep{\fill}}r@{}}
        \large{\textbf{#1}} & \textbf{#2} \\
        \textit{#3} & \textit{#4}
    \end{tabular*}%
    \par\vspace{-2pt}%
}



% Define a compact bullet-point environment for CV items.
% topsep controls the space between the entry heading and the first bullet.

\newenvironment{cvitems}{%
    \begin{itemize}[
        leftmargin=1.5em,
        itemsep=2pt,
        topsep=5pt,
        parsep=0pt,
        partopsep=0pt
    ]
}{%
    \end{itemize}
}

\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

% The authorization line is set in grey to make it visually secondary.

\fancyfoot[C]{%

            \scriptsize
            \textcolor{gray}{%
                I authorize the processing of my personal data in accordance with local law.\\
                Details of ongoing or unpublished research projects are intentionally limited in accordance with research confidentiality and academic integrity standards.
            }%
        }%
    }%
}
